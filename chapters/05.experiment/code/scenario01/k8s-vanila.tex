# -*- mode: ruby -*-
# vi: set ft=ruby :

Vagrant.configure("2") do |config|
  config.vm.box = "ubuntu/jammy64"
  
  # ==========================================
  # 1. CONTROL PLANE (MASTER NODE)
  # ==========================================
  config.vm.define "master" do |master|
    master.vm.hostname = "k8s-master"
    master.vm.network "private_network", ip: "192.168.56.10"
    
    master.vm.provider "virtualbox" do |vb|
      vb.name = "k8s-master-node"
      vb.memory = "4096"
      vb.cpus = 2
      vb.customize ["modifyvm", :id, "--groups", "/k8s-vanilla"]
    end

    # Using K3s to install control plane and stores access token
    master.vm.provision "shell", inline: <<-SHELL
      export DEBIAN_FRONTEND=noninteractive
      apt-get update -y
      
      # Unveils automatically private network name
      FLANNEL_IFACE=$(ip -4 a | grep 192.168.56 | awk '{print $NF}')
      echo "Private network interface detected: $FLANNEL_IFACE"
      
      echo "Install K3s Server (Master)..."
      curl -sfL https://get.k3s.io | INSTALL_K3S_EXEC="--write-kubeconfig-mode 644 --node-ip=192.168.56.10 --flannel-iface=$FLANNEL_IFACE" sh -
      
      echo "Installl Ansible, Git e Helm..."
      apt-get install -y software-properties-common
      add-apt-repository --yes --update ppa:ansible/ansible
      apt-get install -y ansible git curl
      snap install helm --classic
      
      sleep 15
      
      cp /var/lib/rancher/k3s/server/node-token /vagrant/node-token
      
      mkdir -p /root/.kube
      cp /etc/rancher/k3s/k3s.yaml /root/.kube/config
    SHELL
  end

  # ==========================================
  # 2. DATA PLANE (WORKER NODES)
  # ==========================================
  # Configure workers using loop worker1 (final IP .11) and worker2 (final IP .12)
  (1..2).each do |i|
    config.vm.define "worker#{i}" do |worker|
      worker.vm.hostname = "k8s-worker#{i}"
      worker.vm.network "private_network", ip: "192.168.56.1#{i}"
      
      worker.vm.provider "virtualbox" do |vb|
        vb.name = "k8s-worker#{i}-node"
        vb.memory = "4096"
        vb.cpus = 2
        vb.customize ["modifyvm", :id, "--groups", "/k8s-vanilla"]
      end

      # Reads access token generated by master and installs K3s as Agent (Worker)
      worker.vm.provision "shell", inline: <<-SHELL
        export DEBIAN_FRONTEND=noninteractive
        apt-get update -y
        
        # Unveils automatically private network name in this worker
        FLANNEL_IFACE=$(ip -4 a | grep 192.168.56 | awk '{print $NF}')
        echo "Private network interface detected: $FLANNEL_IFACE"
        
        echo "Awaiting Master Node shares token..."
        while [ ! -f /vagrant/node-token ]; do sleep 2; done
        
        TOKEN=$(cat /vagrant/node-token)
        
        echo "Installing K3s Agent (Worker)..."
        curl -sfL https://get.k3s.io | K3S_URL=https://192.168.56.10:6443 K3S_TOKEN=$TOKEN INSTALL_K3S_EXEC="--node-ip=192.168.56.1#{i} --flannel-iface=$FLANNEL_IFACE" sh -
      SHELL
    end
  end
end
